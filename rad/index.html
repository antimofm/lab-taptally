Il fix
# Trova tutti i file con F59E0B
grep -rl "F59E0B\|f59e0b" src/

# Search-replace in tutti i .tsx/.ts/.css
find src/ -type f \( -name "*.tsx" -o -name "*.ts" -o -name "*.css" \) \
  -exec sed -i '' 's/F59E0B/E18C49/gi' {} +

# Verifica — deve tornare vuoto
grep -rl "F59E0B\|f59e0b" src/
